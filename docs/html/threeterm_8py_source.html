174  def propagate(self):
175  """Propagate inputs through relation and calculate new value for the following concept.
176 
177  Returns:
178  - new value for the following concept
179  """
180 
181  sum = 0
182  # Propagate through three-term relation (PDA)
183  for i in range(len(self.previous)):
184  # D = differential component as difference betwen current and last activation value
185  self.dvalues[i] = self.previous[i].value - self.pvalues[i]
186  # P = proportional component as current activation value
187  self.pvalues[i] = self.previous[i].value
188  # A = averaging component as moving average of historical values

191  sum += self.pweights[i] * self.pvalues[i]
192  sum += self.dweights[i] * self.dvalues[i]
193  sum += self.aweights[i] * self.avalues[i]
